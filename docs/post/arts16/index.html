Algorithm

剑指 Offer 15. 二进制中1的个数

  1. 自己的解法：不断的除和求余
  2. 官方解法：与运算
func hammingWeight1(num uint32) int {
	count := 0
	index := 31
	for index >= 0 {
		multi := num / (uint32(math.Pow(2, float64(index))))
		if multi == 1 {
			count++
		}
		num = num % (uint32(math.Pow(2, float64(index))))
		index--
	}
	return count
}

func hammingWeight2(num uint32) int {
	count := 0
	index := 0
	for index < 32 {
		if 1<<index&num > 0 {
			count++
		}
		index++
	}
	return count
}
